Examples
Copy
var sdkCtx = new RuyiSDKContext { endpoint = RuyiSDKContext.Endpoint.Console, EnabledFeatures = RuyiSDK.SDKFeatures.All };
using (var ruyi = RuyiSDK.CreateInstance(sdkCtx))
{
    while (GameIsRunning) { ruyi.Update(); }
}
